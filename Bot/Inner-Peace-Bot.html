async function toggleAudio(btn, enText, saText) { const icon = btn.querySelector('i'); const isCurrentlyPlaying = icon.classList.contains('fa-volume-up'); window.speechSynthesis.cancel(); resetAllSpeakers(); if (isCurrentlyPlaying) return; // 1. Determine content and prepare "Chunks" for natural pausing const textToPlay = audioLang === 'en' ? enText : (saText || enText); const isSanskrit = audioLang === 'sa'; // Create an array of text parts to read with deliberate pauses // If Sanskrit, we replace '||' with a period to force the engine to pause const chunks = isSanskrit ? textToPlay.replace(/\|\|/g, '.').split('.') : [textToPlay]; icon.classList.remove('fa-volume-mute', 'text-zinc-400'); icon.classList.add('fa-volume-up', 'text-amber-500'); // 2. Play chunks sequentially for (let i = 0; i < chunks.length; i++) { if (i > 0) await new Promise(resolve => setTimeout(resolve, 800)); // 800ms pause await new Promise((resolve) => { const msg = new SpeechSynthesisUtterance(chunks[i].trim()); msg.rate = 0.8; // Slightly slower for better clarity msg.lang = isSanskrit ? 'hi-IN' : 'en-US'; msg.onend = resolve; window.speechSynthesis.speak(msg); }); } resetAllSpeakers(); }
